## Endpoints

  <Card>
    <CardHeader>

      <CardDescription>Retrieve all vehicles</CardDescription>
    </CardHeader>
    <CardContent>
      ```bash
      GET https://sw-next-api.vercel.app/api/v1/vehicles
      ```
    </CardContent>
  </Card>

  <Card>
    <CardHeader>
      <CardTitle className="text-lg">Get Single Vehicle</CardTitle>
      <CardDescription>Get vehicle by ID or slug</CardDescription>
    </CardHeader>
    <CardContent>
      ```bash
      GET https://sw-next-api.vercel.app/api/v1/vehicles/14
      GET https://sw-next-api.vercel.app/api/v1/vehicles/snowspeeder
      ```
    </CardContent>
  </Card>
</div>

## Response Example

```json
{
  "entityId": 14,
  "id": "snowspeeder",
  "name": "Snowspeeder",
  "model": "t-47 airspeeder",
  "manufacturer": "Incom corporation",
  "costInCredits": null,
  "length": 4.5,                     // Meters
  "maxAtmospheringSpeed": 650,       // km/h
  "crew": 2,
  "passengers": 0,
  "cargoCapacity": 10,               // Kilograms
  "consumables": "none",
  "vehicleClass": "airspeeder",
  "pilots": [...],                   // Characters who piloted this
  "films": [...]                     // Films featuring this vehicle
}
```

  <TabsList>
    <TabsTrigger value="search">Search</TabsTrigger>
    <TabsTrigger value="class">By Class</TabsTrigger>
    <TabsTrigger value="sort">Sort by Speed</TabsTrigger>
  </TabsList>

  <TabsContent value="search">
    <div className="text-sm text-zinc-600 dark:text-zinc-400 mb-4">

  </TabsContent>

  <TabsContent value="class">
    <div className="text-sm text-zinc-600 dark:text-zinc-400 mb-4">

    </div>
    ```bash
    GET https://sw-next-api.vercel.app/api/v1/vehicles?vehicleClass=airspeeder
    ```
  </TabsContent>

  <TabsContent value="sort">
    <div className="text-sm text-zinc-600 dark:text-zinc-400 mb-4">

    </div>
    ```bash

    ```
  </TabsContent>
</Tabs>
