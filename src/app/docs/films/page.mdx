# Films

## Endpoints

<div className="grid gap-6 md:grid-cols-2 mb-12 sm:mb-24">
  <Card>
    <CardHeader>
      <CardTitle className="text-lg">Get All Films</CardTitle>
      <CardDescription>Retrieve all Star Wars films</CardDescription>
    </CardHeader>
    <CardContent>
      ```bash
      GET https://sw-next-api.vercel.app/api/v1/films
      ```
    </CardContent>
  </Card>

  <Card>
    <CardHeader>
      <CardTitle className="text-lg">Get Single Film</CardTitle>
      <CardDescription>Get film by ID or slug</CardDescription>
    </CardHeader>
    <CardContent>
      ```bash
      GET https://sw-next-api.vercel.app/api/v1/films/1
      GET https://sw-next-api.vercel.app/api/v1/films/a-new-hope
      ```
    </CardContent>
  </Card>
</div>

## Common Queries

<Tabs defaultValue="sort" className="mb-12 sm:mb-24">
  <TabsList>
    <TabsTrigger value="sort">By Episode</TabsTrigger>
    <TabsTrigger value="release">By Release</TabsTrigger>
    <TabsTrigger value="expand">With Characters</TabsTrigger>
  </TabsList>

  <TabsContent value="sort">
    <div className="text-sm text-zinc-600 dark:text-zinc-400 mb-4">
      Get films in episode order
    </div>
    ```bash
    GET https://sw-next-api.vercel.app/api/v1/films?sort=episode
    ```
  </TabsContent>

  <TabsContent value="release">
    <div className="text-sm text-zinc-600 dark:text-zinc-400 mb-4">
      Get films in release order
    </div>
    ```bash
    GET https://sw-next-api.vercel.app/api/v1/films?sort=release_year
    ```
  </TabsContent>

  <TabsContent value="expand">
    <div className="text-sm text-zinc-600 dark:text-zinc-400 mb-4">
      Get film with full character data
    </div>
    ```bash
    GET https://sw-next-api.vercel.app/api/v1/films/1?expand=characters
    ```
  </TabsContent>
</Tabs>
